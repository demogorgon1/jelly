16 // Do a compaction of just 2 stores
17 template <typename _KeyType, typename _ItemType>
18 size_t
19 PerformOnTwoStores(
20 IHost* aHost,
21 uint32_t aNodeId,
22 FileStatsContext* aStoreFileStatsContext,
23 uint32_t aOldestStoreId,
24 uint32_t aNewStoreId,
25 uint32_t aStoreId1,
26 uint32_t aStoreId2,
27 CompactionResult<_KeyType>* aOut)
28 {
29 // Stores are always written in ascendening key order, so merging them is easy
30 std::unique_ptr<IFileStreamReader> f1(aHost->ReadStoreStream(aNodeId, aStoreId1, aStoreFileStatsContext));
31 std::unique_ptr<IFileStreamReader> f2(aHost->ReadStoreStream(aNodeId, aStoreId2, aStoreFileStatsContext));
32
33 if(!f1 || !f2)
34 return 0; // Could be that the stores no longer exists, just don't do anything then
35
36 {
37 std::unique_ptr<IStoreWriter> fOut(aHost->CreateStore(aNodeId, aNewStoreId, aStoreFileStatsContext));
38 JELLY_CHECK(fOut, Exception::ERROR_COMPACTION_FAILED_TO_OPEN_OUTPUT_STORE, "NewStoreId=%u", aNewStoreId);
39
40 _ItemType item1;
41 bool hasItem1 = false;
42
43 _ItemType item2;
44 bool hasItem2 = false;
45
46 for (;;)
47 {
48 if (!hasItem1)
49 hasItem1 = item1.CompactionRead(f1.get());
50
51 if (!hasItem2)
52 hasItem2 = item2.CompactionRead(f2.get());
53
54 if (!hasItem1 && !hasItem2)
55 break;
56
57 if (hasItem1 && !hasItem2)
58 {
59 uint64_t offset = item1.CompactionWrite(aOldestStoreId, fOut.get());
60 if(offset != UINT64_MAX)
61 aOut->AddItem(item1.GetKey(), item1.GetSeq(), aNewStoreId, offset);
62
63 item1.Reset();
64 hasItem1 = false;
65 }
66 else if (!hasItem1 && hasItem2)
67 {
68 uint64_t offset = item2.CompactionWrite(aOldestStoreId, fOut.get());
69 if (offset != UINT64_MAX)
70 aOut->AddItem(item2.GetKey(), item2.GetSeq(), aNewStoreId, offset);
71
72 item2.Reset();
73 hasItem2 = false;
74 }
75 else
76 {
77 JELLY_ASSERT(hasItem1 && hasItem2);
78
79 if (item1.GetKey() < item2.GetKey())
80 {
81 uint64_t offset = item1.CompactionWrite(aOldestStoreId, fOut.get());
82 if (offset != UINT64_MAX)
83 aOut->AddItem(item1.GetKey(), item1.GetSeq(), aNewStoreId, offset);
84
85 item1.Reset();
86 hasItem1 = false;
87 }
88 else if (item2.GetKey() < item1.GetKey())
89 {
90 uint64_t offset = item2.CompactionWrite(aOldestStoreId, fOut.get());
91 if (offset != UINT64_MAX)
92 aOut->AddItem(item2.GetKey(), item2.GetSeq(), aNewStoreId, offset);
93
94 item2.Reset();
95 hasItem2 = false;
96 }
97 else
98 {
99 // Items are the same - keep the one with the highest sequence number
100 size_t offset = UINT64_MAX;
101 uint32_t seq = 0;
102
103 if (item1.GetSeq() > item2.GetSeq())
104 {
105 seq = item1.GetSeq();
106 offset = item1.CompactionWrite(aOldestStoreId, fOut.get());
107 }
108 else
109 {
110 seq = item2.GetSeq();
111 offset = item2.CompactionWrite(aOldestStoreId, fOut.get());
112 }
113
114 if (offset != UINT64_MAX)
115 aOut->AddItem(item1.GetKey(), seq, aNewStoreId, offset);
116
117 hasItem1 = false;
118 hasItem2 = false;
119
120 item1.Reset();
121 item2.Reset();
122 }
123 }
124 }
125
126 fOut->Flush();
127 }
128
129 return 2;
130 }

132 // Do a compaction of more than 2 stores
133 template <typename _KeyType, typename _ItemType>
134 size_t
135 PerformOnStoreList(
136 IHost* aHost,
137 uint32_t aNodeId,
138 FileStatsContext* aStoreFileStatsContext,
139 uint32_t aOldestStoreId,
140 uint32_t aNewStoreId,
141 const std::vector<uint32_t>& aStoreIds,
142 CompactionResult<_KeyType>* aOut)
143 {
144 JELLY_ASSERT(aStoreIds.size() > 2);
145
146 struct SourceStore
147 {
148 SourceStore(
149 uint32_t aStoreId,
150 IFileStreamReader* aFileStreamReader)
151 : m_hasItem(false)
152 , m_storeId(aStoreId)
153 , m_fileStreamReader(aFileStreamReader)
154 {
155 }
156
157 uint32_t m_storeId;
158
159 std::unique_ptr<IFileStreamReader> m_fileStreamReader;
160
161 _ItemType m_item;
162 bool m_hasItem;
163 };
164
165 std::vector<std::unique_ptr<SourceStore>> sourceStores;
166
167 // Open source stores
168 for (uint32_t storeId : aStoreIds)
169 {
170 std::unique_ptr<SourceStore> sourceStore = std::make_unique<SourceStore>(storeId, aHost->ReadStoreStream(aNodeId, storeId, aStoreFileStatsContext));
171
172 if(sourceStore->m_fileStreamReader)
173 sourceStores.push_back(std::move(sourceStore));
174 }
175
176 if(sourceStores.size() <= 2)
177 return 0;
178
179 // Open output store
180 std::unique_ptr<IStoreWriter> outputStore(aHost->CreateStore(aNodeId, aNewStoreId, aStoreFileStatsContext));
181 JELLY_CHECK(outputStore, Exception::ERROR_COMPACTION_FAILED_TO_OPEN_OUTPUT_STORE, "NewStoreId=%u", aNewStoreId);
182
183 // Perform the compaction
184 std::vector<SourceStore*> lowestKeySourceStores;
185
186 for (;;)
187 {
188 size_t numSourceStoresWithItemsRemaining = 0;
189
190 for (std::unique_ptr<SourceStore>& sourceStore : sourceStores)
191 {
192 if (!sourceStore->m_hasItem && !sourceStore->m_fileStreamReader->IsEnd())
193 sourceStore->m_hasItem = sourceStore->m_item.CompactionRead(sourceStore->m_fileStreamReader.get());
194
195 if (sourceStore->m_hasItem)
196 numSourceStoresWithItemsRemaining++;
197 }
198
199 if (numSourceStoresWithItemsRemaining == 0)
200 break;
201
202 // Find lowest key
203 std::optional<_KeyType> lowestKey;
204
205 for (std::unique_ptr<SourceStore>& sourceStore : sourceStores)
206 {
207 if (sourceStore->m_hasItem && (!lowestKey.has_value() || sourceStore->m_item.GetKey() < lowestKey))
208 lowestKey = sourceStore->m_item.GetKey();
209 }
210
211 JELLY_ASSERT(lowestKey.has_value());
212
213 // Find stores with this key (might be multiple with equal or different sequence numbers)
214 lowestKeySourceStores.clear();
215
216 for (std::unique_ptr<SourceStore>& sourceStore : sourceStores)
217 {
218 if (sourceStore->m_hasItem && sourceStore->m_item.GetKey() == lowestKey)
219 lowestKeySourceStores.push_back(sourceStore.get());
220 }
221
222 JELLY_ASSERT(lowestKeySourceStores.size() > 0);
223
224 // Sort by sequence numbers - we want to use just the latest one
225 std::sort(lowestKeySourceStores.begin(), lowestKeySourceStores.end(), [](
226 const SourceStore* aLHS,
227 const SourceStore* aRHS) -> bool
228 {
229 JELLY_ASSERT(aLHS->m_hasItem && aRHS->m_hasItem);
230 JELLY_ASSERT(aLHS->m_item.GetKey() == aRHS->m_item.GetKey());
231 return aLHS->m_item.GetSeq() > aRHS->m_item.GetSeq();
232 });
233
234 // Verify that array is ordered largest to smallest
235 JELLY_ASSERT(lowestKeySourceStores[0]->m_item.GetSeq() >= lowestKeySourceStores[lowestKeySourceStores.size() - 1]->m_item.GetSeq());
236
237 {
238 _ItemType& sourceItem = lowestKeySourceStores[0]->m_item;
239
240 size_t offset = sourceItem.CompactionWrite(aOldestStoreId, outputStore.get());
241
242 if(offset != UINT64_MAX)
243 aOut->AddItem(lowestKey.value(), sourceItem.GetSeq(), aNewStoreId, offset);
244
245 for (SourceStore* lowestKeyStore : lowestKeySourceStores)
246 {
247 JELLY_ASSERT(lowestKeyStore->m_hasItem);
248 lowestKeyStore->m_hasItem = false;
249
250 lowestKeyStore->m_item.Reset();
251 }
252 }
253 }
254
255 outputStore->Flush();
256
257 return sourceStores.size();
258 }

260 // Do a "minor" compaction
261 template <typename _KeyType, typename _ItemType>
262 size_t
263 Perform(
264 IHost* aHost,
265 uint32_t aNodeId,
266 FileStatsContext* aStoreFileStatsContext,
267 uint32_t aNewStoreId,
268 const CompactionJob& aCompactionJob,
269 CompactionResult<_KeyType>* aOut)
270 {
271 aOut->SetStoreIds(aCompactionJob.m_storeIds);
272
273 if(aCompactionJob.m_storeIds.size() == 2)
274 return PerformOnTwoStores<_KeyType, _ItemType>(aHost, aNodeId, aStoreFileStatsContext, aCompactionJob.m_oldestStoreId, aNewStoreId, aCompactionJob.m_storeIds[0], aCompactionJob.m_storeIds[1], aOut);
275 else
276 return PerformOnStoreList<_KeyType, _ItemType>(aHost, aNodeId, aStoreFileStatsContext, aCompactionJob.m_oldestStoreId, aNewStoreId, aCompactionJob.m_storeIds, aOut);
277 }

279 // Do a "major" compaction of everything.
280 template <typename _KeyType, typename _ItemType>
281 size_t
282 PerformMajorCompaction(
283 IHost* aHost,
284 uint32_t aNodeId,
285 FileStatsContext* aStoreFileStatsContext,
286 uint32_t aNewStoreId,
287 CompactionResult<_KeyType>* aOut)
288 {
289 // Enumerate all stores
290 std::vector<IHost::StoreInfo> storeInfo;
291 aHost->GetStoreInfo(aNodeId, storeInfo);
292
293 // Need at least 3 stores for compaction (can't touch the newest one)
294 if (storeInfo.size() >= 3)
295 {
296 uint32_t oldestStoreId = storeInfo[0].m_id;
297
298 std::vector<uint32_t> storeIds;
299 for(size_t i = 0; i < storeInfo.size() - 1; i++)
300 storeIds.push_back(storeInfo[i].m_id);
301
302 aOut->SetStoreIds(storeIds);
303
304 if (storeIds.size() == 2)
305 return PerformOnTwoStores<_KeyType, _ItemType>(aHost, aNodeId, aStoreFileStatsContext, oldestStoreId, aNewStoreId, storeIds[0], storeIds[1], aOut);
306 else
307 return PerformOnStoreList<_KeyType, _ItemType>(aHost, aNodeId, aStoreFileStatsContext, oldestStoreId, aNewStoreId, storeIds, aOut);
308 }
309
310 return 0;
311 }
