255 RequestResult
256 _Update(
257 Request* aRequest,
258 bool aDelete)
259 {
260 bool obeyResidentBlobSizeLimit = false;
261
262 std::unique_ptr<IBuffer> newBlob;
263 if(!aDelete)
264 {
265 size_t requestBlobSize = aRequest->GetBlob()->GetSize();
266
267 const Compression::IProvider* compression = this->m_host->GetCompressionProvider();
268 if(compression != NULL)
269 newBlob.reset(compression->CompressBuffer(aRequest->GetBlob()));
270 else
271 newBlob.reset(aRequest->DetachBlob());
272
273 this->m_host->GetStats()->Emit(Stat::ID_UNCOMPRESSED_BLOB_SIZE, requestBlobSize);
274 this->m_host->GetStats()->Emit(Stat::ID_COMPRESSED_BLOB_SIZE, newBlob->GetSize());
275 }
276
277 uint32_t requestSeq = aRequest->GetSeq();
278
279 std::pair<Item*, bool> result = this->m_table.InsertOrUpdate(aRequest->GetKey(), [aRequest, aDelete, requestSeq]()
280 {
281 if(aDelete)
282 return (Item*)NULL;
283 else
284 return new Item(aRequest->GetKey(), requestSeq);
285 });
286
287 Item* item = result.first;
288
289 if(result.second)
290 {
291 // This is an insert.
292
293 if(aDelete)
294 {
295 return REQUEST_RESULT_DOES_NOT_EXIST;
296 }
297 else
298 {
299 item->SetBlob(newBlob.release());
300
301 m_totalResidentBlobSize += item->GetBlob()->GetSize();
302 m_residentItems.Add(item);
303
304 obeyResidentBlobSizeLimit = true;
305 }
306 }
307 else
308 {
309 // This is an update.
310
311 if(_AutoIncrementSeq)
312 requestSeq = item->GetSeq() + 1;
313
314 if(item->GetSeq() >= requestSeq)
315 {
316 // Trying to set an old version - return latest sequence number to requester
317 aRequest->SetSeq(item->GetSeq());
318 return REQUEST_RESULT_OUTDATED;
319 }
320
321 if(aRequest->GetLockSeq().has_value() && item->GetLockSeq() != aRequest->GetLockSeq().value())
322 {
323 // We don't have the lock
324 return REQUEST_RESULT_NOT_LOCKED;
325 }
326
327 if(item->HasBlob())
328 {
329 // Already has a blob and is resident
330 JELLY_ASSERT(m_totalResidentBlobSize >= item->GetBlob()->GetSize());
331 m_totalResidentBlobSize -= item->GetBlob()->GetSize();
332
333 if(!aDelete)
334 m_totalResidentBlobSize += newBlob->GetSize();
335
336 m_residentItems.MoveToTail(item);
337 }
338 else if(item->HasTombstone())
339 {
340 // Was deleted (and is also resident)
341 if (!aDelete)
342 m_totalResidentBlobSize += newBlob->GetSize();
343
344 m_residentItems.MoveToTail(item);
345
346 obeyResidentBlobSizeLimit = true;
347 }
348 else
349 {
350 // No blob, not resident
351 if(!aDelete)
352 m_totalResidentBlobSize += newBlob->GetSize();
353
354 m_residentItems.Add(item);
355
356 obeyResidentBlobSizeLimit = true;
357 }
358
359 if(aDelete)
360 item->SetBlob(NULL);
361 else
362 item->SetBlob(newBlob.release());
363 }
364
365 item->GetRuntimeState().m_isResident = true;
366
367 item->SetSeq(requestSeq);
368 item->SetTimeStamp(aRequest->GetTimeStamp());
369
370 if(aRequest->GetMeta().has_value())
371 item->SetMeta(aRequest->GetMeta().value());
372
373 if(aDelete)
374 item->SetTombstoneStoreId(this->GetNextStoreId());
375 else
376 item->RemoveTombstone();
377
378 if(!aRequest->IsNoWrite())
379 {
380 aRequest->WriteToWAL(this, item);
381
382 if (obeyResidentBlobSizeLimit)
383 _ObeyResidentBlobLimits();
384 }
385
386 return REQUEST_RESULT_OK;
387 }

389 RequestResult
390 _Get(
391 Request* aRequest)
392 {
393 Item* item = this->m_table.Get(aRequest->GetKey());
394 if(item == NULL)
395 return REQUEST_RESULT_DOES_NOT_EXIST;
396
397 if(item->HasTombstone())
398 return REQUEST_RESULT_DOES_NOT_EXIST;
399
400 if(item->GetSeq() < aRequest->GetSeq())
401 {
402 // Return stored sequence number
403 aRequest->SetSeq(item->GetSeq());
404 return REQUEST_RESULT_OUTDATED;
405 }
406
407 if(!item->HasBlob())
408 {
409 typename Item::RuntimeState& runtimeState = item->GetRuntimeState();
410
411 JELLY_ASSERT(runtimeState.m_storeId != UINT32_MAX);
412
413 uint32_t storeId = runtimeState.m_storeId;
414 size_t storeOffset = runtimeState.m_storeOffset;
415 IStoreBlobReader* storeBlobReader = this->m_host->GetStoreBlobReader(this->m_nodeId, storeId, &this->m_statsContext.m_fileStore);
416 JELLY_CHECK(storeBlobReader != NULL, Exception::ERROR_FAILED_TO_GET_BLOB_READER, "NodeId=%u;StoreId=%u", this->m_nodeId, storeId);
417
418 storeBlobReader->ReadItemBlob(storeOffset, item);
419
420 m_totalResidentBlobSize += item->GetBlob()->GetSize();
421
422 m_residentItems.Add(item);
423
424 Compression::IProvider* compression = this->m_host->GetCompressionProvider();
425 if(compression != NULL)
426 aRequest->SetBlob(compression->DecompressBuffer(item->GetBlob()));
427 else
428 aRequest->SetBlob(item->GetBlob()->Copy());
429
430 _ObeyResidentBlobLimits();
431 }
432 else
433 {
434 m_residentItems.MoveToTail(item);
435
436 Compression::IProvider* compression = this->m_host->GetCompressionProvider();
437 if (compression != NULL)
438 aRequest->SetBlob(compression->DecompressBuffer(item->GetBlob()));
439 else
440 aRequest->SetBlob(item->GetBlob()->Copy());
441 }
442
443 aRequest->SetSeq(item->GetSeq());
444 aRequest->SetTimeStamp(item->GetTimeStamp());
445 aRequest->SetMeta(item->GetMeta());
446
447 if(aRequest->GetLockSeq().has_value())
448 item->SetLockSeq(aRequest->GetLockSeq().value());
449
450 return REQUEST_RESULT_OK;
451 }

453 void
454 _ReplicateItem(
455 Item& aItem)
456 {
457 JELLY_ASSERT(this->m_replicationNetwork != NULL && !this->m_replicationNetwork->IsLocalNodeMaster());
458
459 const _KeyType& key = aItem.GetKey();
460 uint32_t seq = aItem.GetSeq();
461 bool hasTombstone = aItem.HasTombstone();
462
463 std::pair<Item*, bool> result = this->m_table.InsertOrUpdate(key, [key, seq, hasTombstone]() -> Item*
464 {
465 if(hasTombstone)
466 return (Item*)NULL;
467 else
468 return new Item(key, seq);
469 });
470
471 if(result.first != NULL && (result.second || seq > result.first->GetSeq()))
472 {
473 JELLY_ASSERT(hasTombstone || aItem.HasBlob());
474 Item* existing = result.first;
475
476 if(existing->HasBlob())
477 {
478 JELLY_ASSERT(m_totalResidentBlobSize >= existing->GetBlob()->GetSize());
479 m_totalResidentBlobSize -= existing->GetBlob()->GetSize();
480 m_residentItems.MoveToTail(existing);
481 }
482 else
483 {
484 m_residentItems.Add(existing);
485 }
486
487 m_totalResidentBlobSize += aItem.GetBlob()->GetSize();
488
489 existing->MoveFrom(&aItem);
490 existing->GetRuntimeState().m_isResident = true;
491
492 this->WriteToWAL(result.first, false, NULL);
493
494 _ObeyResidentBlobLimits();
495 }

518 void
519 _Restore()
520 {
521 std::vector<uint32_t> walIds;
522 std::vector<uint32_t> storeIds;
523
524 this->m_host->EnumerateFiles(this->m_nodeId, walIds, storeIds);
525
526 // Set next store id to be +1 of the current highest one
527 if(storeIds.size() > 0)
528 this->SetNextStoreId(storeIds[storeIds.size() - 1] + 1);
529
530 // Load stores in reverse order, newest first. This means that if we hit the memory limit while loading, we'll probably
531 // have the newer data in memory.
532 for (std::vector<uint32_t>::reverse_iterator i = storeIds.rbegin(); i != storeIds.rend(); i++)
533 {
534 uint32_t id = *i;
535
536 std::unique_ptr<IFileStreamReader> f(this->m_host->ReadStoreStream(this->m_nodeId, id, &this->m_statsContext.m_fileStore));
537 if (f)
538 _LoadStore(f.get(), id);
539 }

541 // All stores loaded into memory, sort items by timestamp
542 {
543 struct TimeStampedKey
544 {
545 _KeyType m_key;
546 uint64_t m_timeStamp = 0;
547
548 bool
549 operator<(
550 const TimeStampedKey& aOther) const
551 {
552 if (m_timeStamp == aOther.m_timeStamp)
553 return m_key < aOther.m_key;
554
555 return m_timeStamp < aOther.m_timeStamp;
556 }
557 };
558
559 typedef std::map<TimeStampedKey, Item*> TimeStampSorter;
560 typedef std::pair<const TimeStampedKey, Item*> TimeStampSorterValue;
561
562 TimeStampSorter timeStampSorter;
563 size_t totalSize = 0;
564
565 this->m_table.ForEach([&](
566 Item* aItem) -> bool
567 {
568 if (aItem->HasBlob())
569 {
570 timeStampSorter.insert(TimeStampSorterValue({ aItem->GetKey(), aItem->GetTimeStamp() }, aItem));
571
572 totalSize += aItem->GetBlob()->GetSize();
573 }
574
575 return true;
576 });
577
578 JELLY_ASSERT(totalSize == m_totalResidentBlobSize);
579 JELLY_ASSERT(m_residentItems.IsEmpty());
580
581 for (TimeStampSorterValue& value : timeStampSorter)
582 m_residentItems.Add(value.second);
583 }
584
585 for (uint32_t id : walIds)
586 {
587 WAL* wal = this->AddWAL(id, NULL);
588
589 std::unique_ptr<IFileStreamReader> f(this->m_host->ReadWALStream(this->m_nodeId, id, false, &this->m_statsContext.m_fileWAL));
590 if (f)
591 _LoadWAL(f.get(), wal);
592
593 if (wal->GetRefCount() > 0)
594 this->m_nextWALId = id + 1;
595 }
596
597 this->CleanupWALs();
598
599 _ObeyResidentBlobLimits();

602 void
603 _LoadWAL(
604 IFileStreamReader* aReader,
605 WAL* aWAL)
606 {
607 while (!aReader->IsEnd())
608 {
609 std::unique_ptr<Item> item(new Item());
610 if(!item->Read(aReader, NULL))
611 break;
612
613 typename Item::RuntimeState& itemRuntimeState = item->GetRuntimeState();
614
615 _KeyType key = item.get()->GetKey();
616
617 Item* existing = this->m_table.Get(key);
618 if (existing != NULL)
619 {
620 // We have an existing item for this key - update it if new sequence number is higher
621 if (item->GetSeq() > existing->GetSeq())
622 {
623 if(existing->HasBlob())
624 {
625 JELLY_ASSERT(m_totalResidentBlobSize >= existing->GetBlob()->GetSize());
626
627 m_totalResidentBlobSize -= existing->GetBlob()->GetSize();
628 }
629
630 if(item->HasBlob())
631 m_totalResidentBlobSize += item->GetBlob()->GetSize();
632
633 bool wasResident = existing->GetRuntimeState().m_isResident;
634
635 existing->MoveFrom(item.get());
636
637 // Resident items sorted by age (newest at tail, which is this item now)
638 typename Item::RuntimeState& existingRuntimeState = existing->GetRuntimeState();
639
640 if(wasResident)
641 m_residentItems.MoveToTail(existing);
642 else
643 m_residentItems.Add(existing);
644
645 if (existingRuntimeState.m_pendingWAL != NULL)
646 {
647 // Item is already in pending store list, remove reference to current WAL as we'll add a new one
648 existingRuntimeState.m_pendingWAL->RemoveReference();
649 existingRuntimeState.m_pendingWAL = NULL;
650 }
651 else
652 {
653 // Item isn't in the pending store, add it
654 this->m_pendingStore.insert(std::pair<const _KeyType, Item*>(existing->GetKey(), existing));
655 }
656
657 existingRuntimeState.m_isResident = true;
658 existingRuntimeState.m_pendingWAL = aWAL;
659 existingRuntimeState.m_pendingWAL->AddReference();
660 }
661 }
662 else
663 {
664 // First time we see this key, add it
665 m_residentItems.Add(item.get());
666
667 itemRuntimeState.m_isResident = true;
668 itemRuntimeState.m_pendingWAL = aWAL;
669 itemRuntimeState.m_pendingWAL->AddReference();
670
671 this->m_pendingStore.insert(std::pair<const _KeyType, Item*>(item->GetKey(), item.get()));
672
673 if(item->HasBlob())
674 m_totalResidentBlobSize += item->GetBlob()->GetSize();
675
676 this->m_table.Insert(key, item.release());
677 }
678 }
679
680 aWAL->SetSize(aReader->GetTotalBytesRead());
681 }

683 void
684 _LoadStore(
685 IFileStreamReader* aReader,
686 uint32_t aStoreId)
687 {
688 size_t maxResidentBlobSize = this->m_config.GetSize(Config::ID_MAX_RESIDENT_BLOB_SIZE);
689 size_t maxResidentBlobCount = this->m_config.GetSize(Config::ID_MAX_RESIDENT_BLOB_COUNT);
690
691 while (!aReader->IsEnd())
692 {
693 std::unique_ptr<Item> item(new Item());
694
695 typename Item::RuntimeState& itemRuntimeState = item->GetRuntimeState();
696
697 itemRuntimeState.m_storeOffset = aReader->GetTotalBytesRead();
698 itemRuntimeState.m_storeId = aStoreId;
699
700 if(!item->Read(aReader, &itemRuntimeState.m_storeOffset))
701 break;
702
703 itemRuntimeState.m_storeSize = item->GetBlob()->GetSize();
704
705 _KeyType key = item->GetKey();
706
707 if(m_totalResidentBlobSize >= maxResidentBlobSize || this->GetItemCount() >= maxResidentBlobCount)
708 {
709 item->SetBlob(NULL);
710 itemRuntimeState.m_isResident = false;
711 }
712 else
713 {
714 itemRuntimeState.m_isResident = true;
715 }
716
717 Item* existing = this->m_table.Get(key);
718 if (existing != NULL)
719 {
720 if (item->GetSeq() > existing->GetSeq())
721 {
722 if(existing->HasBlob())
723 {
724 JELLY_ASSERT(m_totalResidentBlobSize >= existing->GetBlob()->GetSize());
725 m_totalResidentBlobSize -= existing->GetBlob()->GetSize();
726 }
727
728 if(item->HasBlob())
729 m_totalResidentBlobSize += item->GetBlob()->GetSize();
730
731 existing->MoveFrom(item.get());
732 }
733 }
734 else
735 {
736 if(item->HasBlob())
737 m_totalResidentBlobSize += item->GetBlob()->GetSize();
738
739 this->m_table.Insert(key, item.release());
740 }
741 }

744 void
745 _ObeyResidentBlobLimits()
746 {
747 ScopedTimeSampler timeSampler(this->m_host->GetStats(), Stat::ID_OBEY_RESIDENT_BLOB_LIMITS_TIME);
748
749 size_t maxResidentBlobSize = this->m_config.GetSize(Config::ID_MAX_RESIDENT_BLOB_SIZE);
750 size_t maxResidentBlobCount = this->m_config.GetSize(Config::ID_MAX_RESIDENT_BLOB_COUNT);
751
752 while(m_totalResidentBlobSize > maxResidentBlobSize || m_residentItems.m_count > maxResidentBlobCount)
753 {
754 Item* head = m_residentItems.m_head;
755 JELLY_ASSERT(head != NULL);
756
757 typename Item::RuntimeState& runtimeState = head->GetRuntimeState();
758
759 JELLY_ASSERT(runtimeState.m_isResident);
760
761 if(runtimeState.m_pendingWAL != NULL)
762 {
763 // We can't expell things that are currently in a WAL
764 break;
765 }
766
767 JELLY_ASSERT(head->HasBlob());
768 m_totalResidentBlobSize -= head->GetBlob()->GetSize();
769 head->SetBlob(NULL);
770 m_residentItems.Remove(head);
771
772 runtimeState.m_isResident = false;
773 }
774
775 this->m_host->GetStats()->Emit(Stat::ID_TOTAL_RESIDENT_BLOB_SIZE, m_totalResidentBlobSize);
776 this->m_host->GetStats()->Emit(Stat::ID_TOTAL_RESIDENT_BLOB_COUNT, m_residentItems.m_count);
777 }
