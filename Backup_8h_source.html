57 {
58 JELLY_CONTEXT(Exception::CONTEXT_BACKUP_PERFORM);
59
60 if(m_compactionJob.IsSet())
61 {
62 JELLY_ASSERT(m_compactionStoreId.has_value());
63 m_compactionResult = std::make_unique<CompactionResultType>();
64
65 size_t compactedCount = Compaction::Perform<_KeyType, _ItemType>(
66 m_host, m_nodeId, m_fileStatsContext, m_compactionStoreId.value(), m_compactionJob, m_compactionResult.get());
67
68 JELLY_CHECK(compactedCount == m_compactionJob.m_storeIds.size(), Exception::ERROR_BACKUP_FAILED_COMPACTION, "Name=%s", m_name.c_str());
69 }
70
71 std::string rootPath = m_backupPath + "/" + m_name;
72
73 // Create the backup directory
74 bool ok = std::filesystem::create_directories(rootPath);
75 JELLY_CHECK(ok, Exception::ERROR_BACKUP_FAILED_TO_CREATE_DIRECTORY, "Name=%s;Path=%s", m_name.c_str(), rootPath.c_str());
76
77 // Put hard links to store files in it
78 for(uint32_t storeId : m_includeStoreIds)
79 {
80 std::string storePath = m_host->GetStorePath(m_nodeId, storeId);
81 std::string storeFileName = StringUtils::GetFileNameFromPath(storePath.c_str());
82 std::string backupStorePath = rootPath + "/" + storeFileName;
83
84 std::error_code errorCode;
85 std::filesystem::create_hard_link(storePath, backupStorePath, errorCode);
86 JELLY_CHECK(!errorCode, Exception::ERROR_BACKUP_FAILED_TO_CREATE_HARD_LINK, "Name=%s;Msg=%s;Path=%s;Backup=%s", m_name.c_str(), errorCode.message().c_str(), storePath.c_str(), backupStorePath.c_str());
87 }
88
89 if(m_prevName.length() > 0)
90 {
91 // Write a little text file with the name of the previous backup that this incremental backup is based on
92 std::string prevPath = rootPath + "/prev.txt";
93 FILE* f = fopen(prevPath.c_str(), "wb");
94 JELLY_CHECK(f != NULL, Exception::ERROR_BACKUP_FAILED_TO_CREATE_PREV_FILE, "Name=%s;Path=%s", m_name.c_str(), prevPath.c_str());
95
96 fprintf(f, "%s\r\n", m_prevName.c_str());
97 fclose(f);
98 }
99
100 m_performed = true;
101 }
