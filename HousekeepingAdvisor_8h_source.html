167 _UpdateConcurrentWALState(
168 EventHandler aEventHandler)
169 {
170 for (size_t i = 0; i < m_concurrentWALState.size(); i++)
171 {
172 ConcurrentWALState& concurrentWALState = m_concurrentWALState[i];

174 size_t pendingRequests = m_node->GetPendingWALRequestCount((uint32_t)i);

176 if(pendingRequests > 0 && concurrentWALState.m_flushCooldown.HasExpired())

215 || totalWALSize > m_config.GetSize(Config::ID_PENDING_STORE_WAL_SIZE_LIMIT)
216 || totalWALCount > m_config.GetSize(Config::ID_PENDING_STORE_WAL_COUNT_LIMIT))
