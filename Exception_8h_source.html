278 inline constexpr Exception::Code
279 MakeExceptionCode(
280 uint32_t aError,
281 uint32_t aContext,
282 uint32_t aRequestType,
283 uint32_t aLogFingerprint) noexcept
284 {
285 return ((aError & ERROR_BIT_MASK) << ERROR_BIT_OFFSET)
286 | ((aContext & CONTEXT_BIT_MASK) << CONTEXT_BIT_OFFSET)
287 | ((aRequestType & REQUEST_TYPE_BIT_MASK) << REQUEST_TYPE_BIT_OFFSET)
288 | ((aLogFingerprint & LOG_FINGERPRINT_BIT_MASK) << LOG_FINGERPRINT_BIT_OFFSET);
289 }
