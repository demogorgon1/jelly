80 BackupType*
81 StartBackup(
82 const char* aName)
83 {
84 JELLY_CONTEXT(Exception::CONTEXT_NODE_START_BACKUP);
85
86 std::string prevName;
87 std::string backupPath = m_config.GetString(Config::ID_BACKUP_PATH);
88 uint32_t latestStoreId = UINT32_MAX;
89 bool incremental = false;
90
91 if(m_config.GetBool(Config::ID_BACKUP_INCREMENTAL))
92 {
93 if(m_host->GetLatestBackupInfo(m_nodeId, backupPath.c_str(), prevName, latestStoreId))
94 incremental = true;
95 }
96
97 std::vector<IHost::StoreInfo> storeInfo;
98 m_host->GetStoreInfo(m_nodeId, storeInfo);
99
100 std::vector<uint32_t> includeStoreIds;
101 size_t totalIncludedStoreSize = 0;
102
103 for(const IHost::StoreInfo& store : storeInfo)
104 {
105 if(!incremental || store.m_id > latestStoreId)
106 {
107 includeStoreIds.push_back(store.m_id);
108
109 totalIncludedStoreSize += store.m_size;
110 }
111 }
112
113 JELLY_CHECK(includeStoreIds.size() > 0, Exception::ERROR_NOTHING_TO_BACKUP, "Incremental=%u", incremental);
114
115 std::unique_ptr<BackupType> backup = std::make_unique<BackupType>(
116 m_host,
117 m_nodeId,
118 backupPath.c_str(),
119 aName,
120 prevName.c_str(),
121 &m_statsContext.m_fileStore);
122
123 if(m_config.GetBool(Config::ID_BACKUP_COMPACTION))
124 {
125 size_t availableDiskSpace = m_host->GetAvailableDiskSpace();
126
127 JELLY_CHECK(availableDiskSpace >= totalIncludedStoreSize, Exception::ERROR_NOT_ENOUGH_AVAILABLE_SPACE_FOR_BACKUP, "Available=%s;MoreRequired=%s",
128 StringUtils::MakeSizeString(availableDiskSpace).c_str(),
129 StringUtils::MakeSizeString(totalIncludedStoreSize - availableDiskSpace).c_str());
130
131 uint32_t compactionStoreId = CreateStoreId();
132
133 backup->SetCompactionJob(compactionStoreId, storeInfo[0].m_id, includeStoreIds);
134 backup->SetIncludeStoreIds({ compactionStoreId });
135 }
136 else
137 {
138 backup->SetIncludeStoreIds(includeStoreIds);
139 }
140
141 return backup.release();
142 }

198 void
199 Stop() noexcept
200 {
201 std::vector<_RequestType*> pendingRequests;
202
203 // Flag request queue as stopped, get any pending requests there are
204 {
205 std::lock_guard lock(m_requestsLock);
206
207 if(m_stopped)
208 return;
209
210 m_stopped = true;
211
212 for(size_t i = 0; i < 2; i++)
213 {
214 for (_RequestType* request = m_requests[i].m_first; request != NULL; request = request->GetNext())
215 pendingRequests.push_back(request);
216
217 m_requests[i].Reset();
218 }
219 }
220
221 // Cancel all pending requests
222 for(_RequestType* request : pendingRequests)
223 request->GetCompletion()->OnCancel();
224
225 // Notify WALs
226 for(WAL* wal : m_wals)
227 wal->Cancel();
228 }

234 size_t
235 ProcessRequests()
236 {
237 JELLY_CONTEXT(Exception::CONTEXT_NODE_PROCESS_REQUESTS);
238
239 Queue<_RequestType>* queue = NULL;
240
241 ScopedTimeSampler timeSampler(m_host->GetStats(), m_statsContext.m_idProcessRequestsTime);
242
243 {
244 std::lock_guard lock(m_requestsLock);
245
246 JELLY_ASSERT(!m_stopped);
247
248 queue = &m_requests[m_requestsWriteIndex];
249
250 m_requestsWriteIndex = (m_requestsWriteIndex + 1) % 2;
251 }
252
253 size_t count = queue->m_count;
254
255 queue->SetGuard();
256
257 if (queue->m_first != NULL)
258 {
259 if (m_replicationNetwork == NULL || m_replicationNetwork->IsLocalNodeMaster())
260 {
261 for (_RequestType* request = queue->m_first; request != NULL; request = request->GetNext())
262 request->Execute();
263 }
264 else
265 {
266 for (_RequestType* request = queue->m_first; request != NULL; request = request->GetNext())
267 request->SetResult(REQUEST_RESULT_NOT_MASTER);
268 }
269
270 _RequestType* request = queue->m_first;
271 while (request != NULL)
272 {
273 // We need to store 'next' before signaling completion as the waiting thread might delete the request immediately
274 _RequestType* next = request->GetNext();
275
276 if (!request->HasPendingWrite() || request->GetResult() == REQUEST_RESULT_EXCEPTION)
277 request->GetCompletion()->Signal();
278
279 request = next;
280 }
281 }
282
283 queue->Reset();
284
285 m_host->GetStats()->Emit(m_statsContext.m_idWALCount, m_wals.size());
286
287 return count;
288 }

533 size_t
534 ApplyCompactionResult(
535 CompactionResultType* aCompactionResult)
536 {
537 JELLY_CONTEXT(Exception::CONTEXT_NODE_APPLY_COMPACTION_RESULT);
538
539 ScopedTimeSampler timeSampler(m_host->GetStats(), m_statsContext.m_idApplyCompactionTime);
540
541 size_t updatedItemCount = 0;
542
543 for (const typename CompactionResultType::Item& compactionResultItem : aCompactionResult->GetItems())
544 {
545 _ItemType* item = m_table.Get(compactionResultItem.m_key);
546 JELLY_ASSERT(item != NULL);
547
548 if (compactionResultItem.m_seq == item->GetSeq())
549 {
550 item->CompactionUpdate(compactionResultItem.m_storeId, compactionResultItem.m_storeOffset);
551 updatedItemCount++;
552 }
553 }
554
555 for (uint32_t storeId : aCompactionResult->GetStoreIds())
556 m_host->DeleteStore(m_nodeId, storeId);
557
558 {
559 std::lock_guard lock(m_currentCompactionStoreIdsLock);
560
561 if (m_currentCompactionIsMajor)
562 {
563 JELLY_ASSERT(m_currentCompactionStoreIds.size() == 0);
564
565 m_currentCompactionIsMajor = false;
566 }
567 else
568 {
569 for (uint32_t deletedStoreId : aCompactionResult->GetStoreIds())
570 m_currentCompactionStoreIds.erase(deletedStoreId);
571 }
572 }
573
574 return updatedItemCount;
575 }

596 void
597 WriteToWAL(
598 _ItemType* aItem,
599 bool aLowPrio,
600 Completion* aCompletion)
601 {
602 typename _ItemType::RuntimeState& runtimeState = aItem->GetRuntimeState();
603
604 if (runtimeState.m_pendingWAL != NULL)
605 {
606 runtimeState.m_pendingWAL->RemoveReference();
607 runtimeState.m_pendingWAL = NULL;
608 }
609 else
610 {
611 m_pendingStore.insert(std::pair<const _KeyType, _ItemType*>(aItem->GetKey(), aItem));
612 }
613
614 {
615 std::vector<WAL*>& pendingWALs = aLowPrio ? m_pendingWALsLowPrio : m_pendingWALs;
616
617 uint32_t walConcurrencyIndex = 0;
618
619 // Pick a pending WAL to write to, using the hash of item key
620 {
621 static_assert(sizeof(size_t) >= sizeof(uint64_t));
622 size_t hash = (size_t)aItem->GetKey().GetHash();
623
624 walConcurrencyIndex = (uint32_t)(pendingWALs.size() * (hash >> 32) / 0x100000000);
625 }
626
627 WAL* wal = _GetPendingWAL(walConcurrencyIndex, pendingWALs);
628
629 // Append to WAL
630 wal->GetWriter()->WriteItem(aItem, aCompletion);
631
632 runtimeState.m_pendingWAL = wal;
633 runtimeState.m_pendingWAL->AddReference();
634
635 // Number of instances that this item exists in different pending WALs (if an item is written repeatedly
636 // (which is likely), many copies of it will exist in the same WALs)
637 runtimeState.m_walInstanceCount++;
638
639 // Total number of item instances across all pending WALs. This is a useful metric for deciding when to
640 // flush pending store to disk.
641 m_pendingStoreWALItemCount++;
642 }

784 WAL*
785 _GetPendingWAL(
786 uint32_t aConcurrentWALIndex,
787 std::vector<WAL*>& aPendingWALs)
788 {
789 // Get the pending WAL for the concurrent WAL index - if it's getting too big close it and make a new one
790
791 JELLY_ASSERT((size_t)aConcurrentWALIndex < aPendingWALs.size());
792
793 WAL*& pendingWAL = aPendingWALs[aConcurrentWALIndex];
794
795 if (pendingWAL != NULL)
796 {
797 if (pendingWAL->GetWriter()->HadFailure() || pendingWAL->GetWriter()->GetSize() > m_config.GetSize(Config::ID_WAL_SIZE_LIMIT))
798 {
799 pendingWAL->Close();
800 pendingWAL = NULL;
801 }
802 }
803
804 if (pendingWAL == NULL)
805 {
806 uint32_t id = m_nextWALId++;
807
808 pendingWAL = AddWAL(id, m_host->CreateWAL(m_nodeId, id, _CompressWAL, &m_statsContext.m_fileWAL));
809 }
810
811 return pendingWAL;
812 }

814 size_t
815 _FlushPendingWAL(
816 uint32_t aConcurrentWALIndex,
817 std::vector<WAL*>& aPendingWALs,
818 ReplicationNetwork* aReplicationNetwork)
819 {
820 size_t count = 0;
821
822 if (aConcurrentWALIndex == UINT32_MAX)
823 {
824 // Flush all pending WALs
825 for (WAL* pendingWAL : aPendingWALs)
826 {
827 if (pendingWAL != NULL)
828 {
829 ScopedTimeSampler timeSampler(m_host->GetStats(), m_statsContext.m_idFlushPendingWALTime);
830
831 count += pendingWAL->GetWriter()->Flush(aReplicationNetwork);
832 }
833 }
834 }
835 else
836 {
837 // Flush only specific pending WAL
838 JELLY_ASSERT(aConcurrentWALIndex < m_pendingWALs.size());
839
840 WAL* pendingWAL = aPendingWALs[aConcurrentWALIndex];
841 if (pendingWAL != NULL)
842 {
843 ScopedTimeSampler timeSampler(m_host->GetStats(), m_statsContext.m_idFlushPendingWALTime);
844
845 count += pendingWAL->GetWriter()->Flush(aReplicationNetwork);
846 }
847 }
848
849 return count;
850 }
