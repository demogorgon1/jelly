196 size_t
197 Write(
198 IWriter* aWriter) const override
199 {
200 size_t startOffset = aWriter->GetTotalBytesWritten();
201
202 WriteBase(aWriter);
203 m_key.Write(aWriter);
204 m_meta.Write(aWriter);
205 aWriter->WriteUInt(m_lockSeq);
206
207 size_t blobOffset = 0;
208
209 if(!HasTombstone())
210 {
211 JELLY_ASSERT(m_blob);
212
213 aWriter->WriteUInt<size_t>(m_blob->GetSize());
214
215 blobOffset = aWriter->GetTotalBytesWritten() - startOffset;
216
217 aWriter->Write(m_blob->GetPointer(), m_blob->GetSize());
218 }
219 else
220 {
221 JELLY_ASSERT(!m_blob);
222 }
223
224 return blobOffset;
225 }

227 bool
228 Read(
229 IReader* aReader,
230 size_t* aOutBlobOffset) override
231 {
232 size_t startOffset = aReader->GetTotalBytesRead();
233
234 if(!ReadBase(aReader))
235 return false;
236 if(!m_key.Read(aReader))
237 return false;
238 if(!m_meta.Read(aReader))
239 return false;
240 if(!aReader->ReadUInt(m_lockSeq))
241 return false;
242
243 if(!HasTombstone())
244 {
245 size_t size;
246 if (!aReader->ReadUInt<size_t>(size))
247 return false;
248
249 if (aOutBlobOffset != NULL)
250 (*aOutBlobOffset) += aReader->GetTotalBytesRead() - startOffset;
251
252 std::unique_ptr<IBuffer> blob = std::make_unique<Buffer<1>>();
253 blob->SetSize((size_t)size);
254
255 if (aReader->Read(blob->GetPointer(), blob->GetSize()) != blob->GetSize())
256 return false;
257
258 m_blob = std::move(blob);
259
260 m_runtimeState.m_storeSize = m_blob->GetSize();
261 }
262
263 return true;
264 }
